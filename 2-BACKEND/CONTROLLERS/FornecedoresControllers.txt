BuscarFornecedores(){
    return BuscarFornecedoresRepositorio();
}

AdicionarFornecedor(Fornecedor){
    if(!ValidarFornecedorNome(Fornecedor.nome))
        return erro;
    return CriarFornecedorRepositorio(Fornecedor);
}

AtualizarFornecedor(fornecedorId, Fornecedor){
    if(!ValidarFornecedorNome(Fornecedor.nome))
        return erro;
    return AtualizarFornecedorRepositorio(fornecedorId, Fornecedor);
}

RemoverFornecedor(fornecedorId){
    return RemoverFornecedorRepositorio(fornecedorId);
}
