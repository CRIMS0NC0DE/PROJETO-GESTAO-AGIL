DeveRetornarNomeProdutoEstaOk(ValidarNomeProduto("Café"));

DeveRetornarNomeProdutoEstaErrado(ValidarNomeProduto("   "));

Teste: AoAdicionarProdutoValido_RetornaSucesso()
{
	// Arrange
	produto = { nome: "Café Torrado", categoria: "Bebidas", quantidade: 10, preco: 12.5 }
	mock ValidarNomeProduto(produto.nome) => true
	mock CriarProdutoRepositorio(produto) => retornoSucesso

	// Act
	resultado = AdicionarProduto(produto)

	// Assert
	DeveSer(resultado, retornoSucesso)
}

Teste: AoAdicionarProdutoComNomeInvalido_RetornaErro()
{
	// Arrange
	produto = { nome: "   ", categoria: "Bebidas", quantidade: 5 }
	mock ValidarNomeProduto(produto.nome) => false

	// Act
	resultado = AdicionarProduto(produto)

	// Assert
	DeveSer(resultado, erro)
}

Teste: AdicionarProduto_ChamaValidacaoAntesDoRepositorio()
{
	// Arrange
	produto = { nome: "Pão", categoria: "Padaria", quantidade: 20 }
	spy ValidarNomeProduto
	spy CriarProdutoRepositorio

	// Act
	AdicionarProduto(produto)

	// Assert
	DeveTerSidoChamadoCom(ValidarNomeProduto, produto.nome)
	// Se a validação retornou true, o repositório deve ser chamado
	Se(ValidarNomeProduto.retorno == true)
		DeveTerSidoChamadoCom(CriarProdutoRepositorio, produto)
}
