{

    fornecedor = { nome: "ACME", contato: "João", telefone: "(11)99999-9999", email: "contato@acme.com" }
    mock ValidarFornecedorNome(fornecedor.nome) => true
    mock CriarFornecedorRepositorio(fornecedor) => retornoSucesso

}

Teste: ValidarFornecedor_ChamadaAntesDoRepositorioAoCriar()
{
    // Arrange
    fornecedor = { nome: "ACME" }
    spy ValidarFornecedorNome
    spy CriarFornecedorRepositorio

    // Act
    CriarFornecedor(fornecedor)

    // Assert
    DeveTerSidoChamadoCom(ValidarFornecedorNome, fornecedor.nome)
    Se(ValidarFornecedorNome.retorno == true)
        DeveTerSidoChamadoCom(CriarFornecedorRepositorio, fornecedor)
}
